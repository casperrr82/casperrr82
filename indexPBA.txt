function createTabs() {
    const tabButtons = document.getElementById('tabButtons');
    const tabContents = document.getElementById('tabContents');
    
    for(let i = 0; i < 20; i++) {
        // Create tab button
        const button = document.createElement('button');
        button.className = 'tab-button' + (i === 0 ? ' active' : '');
        button.textContent = `Questions ${i*10 + 1}-${(i+1)*10}`;
        button.onclick = () => showTab(i);
        tabButtons.appendChild(button);

        // Create tab content
        const content = document.createElement('div');
        content.className = 'tab-content' + (i === 0 ? ' active' : '');
        content.id = `tab${i}`;
        
        // Add questions to tab
        const questions = examQuestions[i].questions;
        questions.forEach(q => {
            const questionDiv = createQuestionElement(q);
            content.appendChild(questionDiv);
        });

        tabContents.appendChild(content);
    }
}

function createQuestionElement(question) {
    const div = document.createElement('div');
    div.className = 'question';
    div.innerHTML = `
        <p><strong>Question ${question.id}:</strong> ${question.text}</p>
        <div class="options">
            ${question.options.map((opt, idx) => `
                <div class="option">
                    <input type="radio" 
                           name="q${question.id}" 
                           value="${idx}"
                           onchange="updateAnswer(${question.id-1}, ${idx})"
                    >
                    <label>${opt}</label>
                </div>
            `).join('')}
        </div>
        <div class="explanation" id="exp${question.id}">
            <strong>Explanation:</strong> ${question.explanation}
        </div>
    `;
    return div;
}

function showTab(tabIndex) {
    document.querySelectorAll('.tab-button').forEach(b => b.classList.remove('active'));
    document.querySelectorAll('.tab-content').forEach(c => c.classList.remove('active'));
    
    document.querySelectorAll('.tab-button')[tabIndex].classList.add('active');
    document.getElementById(`tab${tabIndex}`).classList.add('active');
}

function updateAnswer(questionId, answer) {
    currentAnswers[questionId] = answer;
    updateProgress();
}

function updateProgress() {
    const answered = currentAnswers.filter(a => a !== null).length;
    const percentage = (answered / 200) * 100;
    document.getElementById('progressBar').style.width = percentage + '%';
}

function submitExam() {
    let correct = 0;
    const incorrect = [];

    currentAnswers.forEach((answer, index) => {
        const questionSet = Math.floor(index / 10);
        const questionIndex = index % 10;
        const question = examQuestions[questionSet].questions[questionIndex];
        
        if(answer === question.correct) {
            correct++;
        } else {
            incorrect.push({
                question: index + 1,
                chosen: answer,
                correct: question.correct
            });
        }
        
        // Show explanations
        document.getElementById(`exp${index+1}`).style.display = 'block';
    });

    const score = (correct / 200) * 100;
    
    document.getElementById('score').textContent = score.toFixed(2);
    document.getElementById('correct').textContent = correct;
    document.getElementById('results').style.display = 'block';
    
    // Show incorrect answers
    const incorrectDiv = document.getElementById('incorrectAnswers');
    incorrectDiv.innerHTML = '<h3>Incorrect Answers:</h3>' + 
        incorrect.map(i => `
            <p>Question ${i.question}: You chose option ${i.chosen !== null ? String.fromCharCode(65 + i.chosen) : 'none'}, 
               correct answer was ${String.fromCharCode(65 + i.correct)}</p>
        `).join('');
}
